. -> any character except new line
\d -> digit (0-9)
\D -> not digit (0-9)
\w -> word character(a-z,A-Z,0-9,_)
\W -> not a Word Characters
\s -> whitespace
\S -> not whitespace
\b -> word Boundary
\B-> not word Boundary
^ -> beggining of string
$ -> end of string

Quantifiers :
* -> 0 or more
+ -> 1 or more
? -> 0 or One
{3} -> exact number
{3,4} -> range of numbers ( min , max)
